523 void cRBDUtil::CalcCoM(const cRBDModel& model, tVector& out_com, tVector& out_vel)
524 {
525  const Eigen::MatrixXd& joint_mat = model.GetJointMat();
526  const Eigen::MatrixXd& body_defs = model.GetBodyDefs();
527  const Eigen::VectorXd& pose = model.GetPose();
528  const Eigen::VectorXd& vel = model.GetVel();
529 
530  int num_joints = cKinTree::GetNumJoints(joint_mat);
531  out_com.setZero();
532  out_vel.setZero();
533  double total_mass = 0;
534 
535  for (int j = 0; j < num_joints; ++j)
536  {
537  if (cKinTree::IsValidBody(body_defs, j))
538  {
539  tVector local_com = cKinTree::GetBodyLocalCoM(body_defs, j);
540  cSpAlg::tSpTrans world_joint_trans = model.GetSpWorldJointTrans(j);
541 
542  tMatrix joint_world_mat = cSpAlg::TransToMat(cSpAlg::InvTrans(world_joint_trans));
543  tMatrix body_joint_mat = cKinTree::BodyJointTrans(body_defs, j);
544  tMatrix body_world_mat = joint_world_mat * body_joint_mat;
545 
546  tVector attach_pt = cKinTree::GetBodyAttachPt(body_defs, j);
547  attach_pt[3] = 1;
548  attach_pt = body_joint_mat * attach_pt;
549 
550  local_com[3] = 1;
551  tVector world_com = body_world_mat * local_com;
552  world_com[3] = 0;
553 
554  cSpAlg::tSpTrans com_trans = cSpAlg::BuildTrans(world_com);
555  cSpAlg::tSpVec sv = CalcWorldVel(model, j);

557 
558  tVector com_vel = cSpAlg::GetV(sv);
559 
560  double m = cKinTree::GetBodyMass(body_defs, j);
561  out_com += m * world_com;
562  out_vel += m * com_vel;
563  total_mass += m;
564  }
565  }
566 
567  assert(total_mass > 0);
568  out_com /= total_mass;
569  out_vel /= total_mass;
570 }

644 cSpAlg::tSpMat cRBDUtil::BuildMomentInertiaBox(const Eigen::MatrixXd& body_defs, int part_id)

667 cSpAlg::tSpMat cRBDUtil::BuildMomentInertiaCapsule(const Eigen::MatrixXd& body_defs, int part_id)
668 {
669  const cKinTree::tBodyDef& def = body_defs.row(part_id);
670  double mass = cKinTree::GetBodyMass(body_defs, part_id);
671  double r = 0.5 * def(cKinTree::eBodyParam0);
672  double h = def(cKinTree::eBodyParam1);
673 
674  double c_vol = M_PI * r * r * h;
675  double hs_vol = M_PI * 2.0 / 3.0 * r * r * r;
676  double density = mass / (c_vol + 2 * hs_vol);
677  double cm = c_vol * density;
678  double hsm = hs_vol * density;
679 
680  double x = cm*(0.25 * r * r + (1.0 / 12.0) * h * h) +
681  2 * hsm *(0.4 * r * r + (3.0 / 8) * r * h + 0.5 * h * h);
682  double y = (0.5 * cm + 0.8 * hsm) * r * r;
683  double z = x;
684 
685  cSpAlg::tSpMat I = cSpAlg::tSpMat::Zero();
686  I(0, 0) = x;
687  I(1, 1) = y;
688  I(2, 2) = z;
689  I(3, 3) = mass;
690  I(4, 4) = mass;
691  I(5, 5) = mass;
692 
693  return I;

932 cSpAlg::tSpVec cRBDUtil::BuildCjRoot(const Eigen::MatrixXd& joint_mat, const Eigen::VectorXd& q, const Eigen::VectorXd& q_dot, int j)
933 {
934  tQuaternion quat = cKinTree::GetRootRot(joint_mat, q);
935  tVector vel_lin = cKinTree::GetRootVel(joint_mat, q_dot);
936  tVector vel_ang = cKinTree::GetRootAngVel(joint_mat, q_dot);
937  vel_ang[3] = 0;
938 
939  Eigen::VectorXd joint_params;
940  cKinTree::GetJointParams(joint_mat, q_dot, j, joint_params);
941 
942  tMatrix vel_dquat_mat = cMathUtil::BuildQuaternionDiffMat(quat);
943  tVector dq_vec = vel_dquat_mat * vel_ang;
944  tQuaternion dquat = cMathUtil::VecToQuat(dq_vec);
945 
946  tMatrix mat = tMatrix::Identity();
947  mat(0, 0) = 4 * (quat.w() * dquat.w() + quat.x() * dquat.x());
948  mat(1, 1) = 4 * (quat.w() * dquat.w() + quat.y() * dquat.y());
949  mat(2, 2) = 4 * (quat.w() * dquat.w() + quat.z() * dquat.z());
950 
951  mat(1, 0) = 2 * (dquat.x() * quat.y() + quat.x() * dquat.y()
952  - dquat.w() * quat.z() - quat.w() * dquat.z());
953  mat(0, 1) = 2 * (dquat.x() * quat.y() + quat.x() * dquat.y()
954  + dquat.w() * quat.z() + quat.w() * dquat.z());
955 
956  mat(2, 0) = 2 * (dquat.x() * quat.z() + quat.x() * dquat.z()
957  + dquat.w() * quat.y() + quat.w() * dquat.y());
958  mat(0, 2) = 2 * (dquat.x() * quat.z() + quat.x() * dquat.z()
959  - dquat.w() * quat.y() - quat.w() * dquat.y());
960 
961  mat(2, 1) = 2 * (dquat.y() * quat.z() + quat.y() * dquat.z()
962  - dquat.w() * quat.x() - quat.w() * dquat.x());
963  mat(1, 2) = 2 * (dquat.y() * quat.z() + quat.y() * dquat.z()
964  + dquat.w() * quat.x() + quat.w() * dquat.x());
965 
966  cSpAlg::tSpVec cj = cSpAlg::tSpVec::Zero();
967  cj.segment(3, 3) = (mat * vel_lin).segment(0, 3);
968 
969  return cj;
970 }
