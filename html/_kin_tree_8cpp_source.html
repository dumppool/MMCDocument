633 Eigen::VectorXi cKinTree::FindJointChain(const Eigen::MatrixXd& joint_mat, int joint_beg, int joint_end)
634 {
635  Eigen::VectorXi chain;
636 
637  const int max_length = 128;
638  int chain_buffer[max_length];
639  int buffer_idx = 0;
640 
641  if (joint_beg == joint_end)
642  {
643  Eigen::VectorXi chain(1);
644  chain[0] = joint_beg;
645  }
646 
647  int common_ancestor = gInvalidJointID;
648  int curr_id = joint_beg;
649  int end_len = 0;
650  while (curr_id != gInvalidJointID)
651  {
652  chain_buffer[buffer_idx] = curr_id;
653  ++buffer_idx;
654 
655  if (buffer_idx >= max_length)
656  {
657  printf("Exceeded maximum chain length %i\n", max_length);
658  assert(false);
659  return chain;
660  }
661 
662  bool is_ancestor = IsAncestor(joint_mat, joint_end, curr_id, end_len);
663  if (is_ancestor)
664  {
665  common_ancestor = curr_id;
666  break;
667  }
668  else
669  {
670  curr_id = GetParent(joint_mat, curr_id);
671  }
672  }
673 
674  bool found = common_ancestor != gInvalidJointID;
675  // tree should always connected?
676  assert(found);
677 
678  if (found)
679  {
680  chain.resize(buffer_idx + end_len);
681  for (int i = 0; i < buffer_idx; ++i)
682  {
683  chain[i] = chain_buffer[i];
684  }
685 
686  int idx = buffer_idx;
687  curr_id = joint_end;
688  while (curr_id != common_ancestor)
689  {
690  chain[idx] = curr_id;
691  curr_id = GetParent(joint_mat, curr_id);
692  ++idx;
693  }
694 
695  int num_flips = static_cast<int>(chain.size()) - buffer_idx;
696  chain.block(buffer_idx, 0, num_flips, 1).reverseInPlace();
697  }
698 
699  return chain;
700 }
