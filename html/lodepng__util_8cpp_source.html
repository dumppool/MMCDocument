161 unsigned getFilterTypesInterlaced(std::vector<std::vector<unsigned char> >& filterTypes,
162  const std::vector<unsigned char>& png)
163 {
164  //Get color type and interlace type
165  lodepng::State state;
166  unsigned w, h;
167  unsigned error;
168  error = lodepng_inspect(&w, &h, &state, &png[0], png.size());
169 
170  if(error) return 1;
171 
172  //Read literal data from all IDAT chunks
173  const unsigned char *chunk, *begin, *end, *next;
174  end = &png.back() + 1;
175  begin = chunk = &png.front() + 8;
176 
177  std::vector<unsigned char> zdata;
178 
179  while(chunk + 8 < end && chunk >= begin)
180  {
181  char type[5];
182  lodepng_chunk_type(type, chunk);
183  if(std::string(type).size() != 4) return 1; //Probably not a PNG file
184 
185  if(std::string(type) == "IDAT")
186  {
187  const unsigned char* cdata = lodepng_chunk_data_const(chunk);
188  unsigned clength = lodepng_chunk_length(chunk);
189  if(chunk + clength + 12 > end || clength > png.size() || chunk + clength + 12 < begin) {
190  // corrupt chunk length
191  return 1;
192  }
193 
194  for(unsigned i = 0; i < clength; i++)
195  {
196  zdata.push_back(cdata[i]);
197  }
198  }
199 
200  next = lodepng_chunk_next_const(chunk);
201  if (next <= chunk) return 1; // integer overflow
202  chunk = next;
203  }
204 
205  //Decompress all IDAT data
206  std::vector<unsigned char> data;
207  error = lodepng::decompress(data, &zdata[0], zdata.size());
208 
209  if(error) return 1;
210 
211  if(state.info_png.interlace_method == 0)
212  {
213  filterTypes.resize(1);
214 
215  //A line is 1 filter byte + all pixels
216  size_t linebytes = 1 + lodepng_get_raw_size(w, 1, &state.info_png.color);
217 
218  for(size_t i = 0; i < data.size(); i += linebytes)
219  {
220  filterTypes[0].push_back(data[i]);
221  }
222  }
223  else
224  {
225  //Interlaced
226  filterTypes.resize(7);
227  static const unsigned ADAM7_IX[7] = { 0, 4, 0, 2, 0, 1, 0 }; /*x start values*/
228  static const unsigned ADAM7_IY[7] = { 0, 0, 4, 0, 2, 0, 1 }; /*y start values*/
229  static const unsigned ADAM7_DX[7] = { 8, 8, 4, 4, 2, 2, 1 }; /*x delta values*/
230  static const unsigned ADAM7_DY[7] = { 8, 8, 8, 4, 4, 2, 2 }; /*y delta values*/
231  size_t pos = 0;
232  for(size_t j = 0; j < 7; j++)
233  {
234  unsigned w2 = (w - ADAM7_IX[j] + ADAM7_DX[j] - 1) / ADAM7_DX[j];
235  unsigned h2 = (h - ADAM7_IY[j] + ADAM7_DY[j] - 1) / ADAM7_DY[j];
236  if(ADAM7_IX[j] >= w) w2 = 0;
237  if(ADAM7_IY[j] >= h) h2 = 0;
238  size_t linebytes = 1 + lodepng_get_raw_size(w2, 1, &state.info_png.color);
239  for(size_t i = 0; i < h2; i++)
240  {
241  filterTypes[j].push_back(data[pos]);
242  pos += linebytes;
243  }
244  }
245  }
246  return 0; /* OK */
247 }

320  struct HuffmanTree
321  {
322  int makeFromLengths(const std::vector<unsigned long>& bitlen, unsigned long maxbitlen)
323  { //make tree given the lengths
324  unsigned long numcodes = (unsigned long)(bitlen.size()), treepos = 0, nodefilled = 0;
325  std::vector<unsigned long> tree1d(numcodes), blcount(maxbitlen + 1, 0), nextcode(maxbitlen + 1, 0);
326  //count number of instances of each code length
327  for(unsigned long bits = 0; bits < numcodes; bits++) blcount[bitlen[bits]]++;
328  for(unsigned long bits = 1; bits <= maxbitlen; bits++)
329  {
330  nextcode[bits] = (nextcode[bits - 1] + blcount[bits - 1]) << 1;
331  }
332  //generate all the codes
333  for(unsigned long n = 0; n < numcodes; n++) if(bitlen[n] != 0) tree1d[n] = nextcode[bitlen[n]]++;
334  tree2d.clear(); tree2d.resize(numcodes * 2, 32767); //32767 here means the tree2d isn't filled there yet
335  for(unsigned long n = 0; n < numcodes; n++) //the codes
336  for(unsigned long i = 0; i < bitlen[n]; i++) //the bits for this code
337  {
338  unsigned long bit = (tree1d[n] >> (bitlen[n] - i - 1)) & 1;
339  if(treepos > numcodes - 2) return 55;
340  if(tree2d[2 * treepos + bit] == 32767) //not yet filled in
341  {
342  if(i + 1 == bitlen[n])
343  {
344  //last bit
345  tree2d[2 * treepos + bit] = n;
346  treepos = 0;
347  }
348  else
349  {
350  //addresses are encoded as values > numcodes
351  tree2d[2 * treepos + bit] = ++nodefilled + numcodes;
352  treepos = nodefilled;
353  }
354  }
355  else treepos = tree2d[2 * treepos + bit] - numcodes; //subtract numcodes from address to get address value
356  }
357  return 0;
358  }
359  int decode(bool& decoded, unsigned long& result, size_t& treepos, unsigned long bit) const
360  { //Decodes a symbol from the tree
361  unsigned long numcodes = (unsigned long)tree2d.size() / 2;
362  if(treepos >= numcodes) return 11; //error: you appeared outside the codetree
363  result = tree2d[2 * treepos + bit];
364  decoded = (result < numcodes);
365  treepos = decoded ? 0 : result - numcodes;
366  return 0;
367  }
368  //2D representation of a huffman tree: one dimension is "0" or "1", the other contains all nodes and leaves.
369  std::vector<unsigned long> tree2d;
370  };

373  {

419  void getTreeInflateDynamic(HuffmanTree& tree, HuffmanTree& treeD,
420  const unsigned char* in, size_t& bp, size_t inlength)
421  {
422  size_t bpstart = bp;
423  //get the tree of a deflated block with dynamic tree, the tree itself is also Huffman compressed with a known tree
424  std::vector<unsigned long> bitlen(288, 0), bitlenD(32, 0);
425  if(bp >> 3 >= inlength - 2) { error = 49; return; } //the bit pointer is or will go past the memory
426  size_t HLIT = readBitsFromStream(bp, in, 5) + 257; //number of literal/length codes + 257
427  size_t HDIST = readBitsFromStream(bp, in, 5) + 1; //number of dist codes + 1
428  size_t HCLEN = readBitsFromStream(bp, in, 4) + 4; //number of code length codes + 4
429  zlibinfo->back().hlit = HLIT - 257;
430  zlibinfo->back().hdist = HDIST - 1;
431  zlibinfo->back().hclen = HCLEN - 4;
432  std::vector<unsigned long> codelengthcode(19); //lengths of tree to decode the lengths of the dynamic tree
433  for(size_t i = 0; i < 19; i++) codelengthcode[CLCL[i]] = (i < HCLEN) ? readBitsFromStream(bp, in, 3) : 0;
434  //code length code lengths
435  for(size_t i = 0; i < codelengthcode.size(); i++) zlibinfo->back().clcl.push_back(codelengthcode[i]);
436  error = codelengthcodetree.makeFromLengths(codelengthcode, 7); if(error) return;
437  size_t i = 0, replength;
438  while(i < HLIT + HDIST)
439  {
440  unsigned long code = huffmanDecodeSymbol(in, bp, codelengthcodetree, inlength); if(error) return;
441  zlibinfo->back().treecodes.push_back(code); //tree symbol code
442  if(code <= 15) { if(i < HLIT) bitlen[i++] = code; else bitlenD[i++ - HLIT] = code; } //a length code
443  else if(code == 16) //repeat previous
444  {
445  if(bp >> 3 >= inlength) { error = 50; return; } //error, bit pointer jumps past memory
446  replength = 3 + readBitsFromStream(bp, in, 2);
447  unsigned long value; //set value to the previous code
448  if((i - 1) < HLIT) value = bitlen[i - 1];
449  else value = bitlenD[i - HLIT - 1];
450  for(size_t n = 0; n < replength; n++) //repeat this value in the next lengths
451  {
452  if(i >= HLIT + HDIST) { error = 13; return; } //error: i is larger than the amount of codes
453  if(i < HLIT) bitlen[i++] = value; else bitlenD[i++ - HLIT] = value;
454  }
455  }
456  else if(code == 17) //repeat "0" 3-10 times
457  {
458  if(bp >> 3 >= inlength) { error = 50; return; } //error, bit pointer jumps past memory
459  replength = 3 + readBitsFromStream(bp, in, 3);
460  zlibinfo->back().treecodes.push_back(replength); //tree symbol code repetitions
461  for(size_t n = 0; n < replength; n++) //repeat this value in the next lengths
462  {
463  if(i >= HLIT + HDIST) { error = 14; return; } //error: i is larger than the amount of codes
464  if(i < HLIT) bitlen[i++] = 0; else bitlenD[i++ - HLIT] = 0;
465  }
466  }
467  else if(code == 18) //repeat "0" 11-138 times
468  {
469  if(bp >> 3 >= inlength) { error = 50; return; } //error, bit pointer jumps past memory
470  replength = 11 + readBitsFromStream(bp, in, 7);
471  zlibinfo->back().treecodes.push_back(replength); //tree symbol code repetitions
472  for(size_t n = 0; n < replength; n++) //repeat this value in the next lengths
473  {
474  if(i >= HLIT + HDIST) { error = 15; return; } //error: i is larger than the amount of codes
475  if(i < HLIT) bitlen[i++] = 0; else bitlenD[i++ - HLIT] = 0;
476  }
477  }
478  else { error = 16; return; } //error: somehow an unexisting code appeared. This can never happen.
479  }
480  if(bitlen[256] == 0) { error = 64; return; } //the length of the end code 256 must be larger than 0
481  error = tree.makeFromLengths(bitlen, 15);
482  if(error) return; //now we've finally got HLIT and HDIST, so generate the code trees, and the function is done
483  error = treeD.makeFromLengths(bitlenD, 15);
484  if(error) return;
485  zlibinfo->back().treebits = bp - bpstart;
486  //lit/len/end symbol lengths
487  for(size_t j = 0; j < bitlen.size(); j++) zlibinfo->back().litlenlengths.push_back(bitlen[j]);
488  //dist lengths
489  for(size_t j = 0; j < bitlenD.size(); j++) zlibinfo->back().distlengths.push_back(bitlenD[j]);
490  }

492  void inflateHuffmanBlock(std::vector<unsigned char>& out,
493  const unsigned char* in, size_t& bp, size_t& pos, size_t inlength, unsigned long btype)
494  {
495  size_t numcodes = 0, numlit = 0, numlen = 0; //for logging
496  if(btype == 1) { generateFixedTrees(codetree, codetreeD); }
497  else if(btype == 2) { getTreeInflateDynamic(codetree, codetreeD, in, bp, inlength); if(error) return; }
498  for(;;)
499  {
500  unsigned long code = huffmanDecodeSymbol(in, bp, codetree, inlength); if(error) return;
501  numcodes++;
502  zlibinfo->back().lz77_lcode.push_back(code); //output code
503  zlibinfo->back().lz77_dcode.push_back(0);
504  zlibinfo->back().lz77_lbits.push_back(0);
505  zlibinfo->back().lz77_dbits.push_back(0);
506  zlibinfo->back().lz77_lvalue.push_back(0);
507  zlibinfo->back().lz77_dvalue.push_back(0);
508 
509  if(code == 256) break; //end code
510  else if(code <= 255) //literal symbol
511  {
512  out.push_back((unsigned char)(code));
513  pos++;
514  numlit++;
515  }
516  else if(code >= 257 && code <= 285) //length code
517  {
518  size_t length = LENBASE[code - 257], numextrabits = LENEXTRA[code - 257];
519  if((bp >> 3) >= inlength) { error = 51; return; } //error, bit pointer will jump past memory
520  length += readBitsFromStream(bp, in, numextrabits);
521  unsigned long codeD = huffmanDecodeSymbol(in, bp, codetreeD, inlength); if(error) return;
522  if(codeD > 29) { error = 18; return; } //error: invalid dist code (30-31 are never used)
523  unsigned long dist = DISTBASE[codeD], numextrabitsD = DISTEXTRA[codeD];
524  if((bp >> 3) >= inlength) { error = 51; return; } //error, bit pointer will jump past memory
525  dist += readBitsFromStream(bp, in, numextrabitsD);
526  size_t start = pos, back = start - dist; //backwards
527  for(size_t i = 0; i < length; i++)
528  {
529  out.push_back(out[back++]);
530  pos++;
531  if(back >= start) back = start - dist;
532  }
533  numlen++;
534  zlibinfo->back().lz77_dcode.back() = codeD; //output distance code
535  zlibinfo->back().lz77_lbits.back() = numextrabits; //output length extra bits
536  zlibinfo->back().lz77_dbits.back() = numextrabitsD; //output dist extra bits
537  zlibinfo->back().lz77_lvalue.back() = length; //output length
538  zlibinfo->back().lz77_dvalue.back() = dist; //output dist
539  }
540  }
541  zlibinfo->back().numlit = numlit; //output number of literal symbols
542  zlibinfo->back().numlen = numlen; //output number of length symbols
543  }

545  void inflateNoCompression(std::vector<unsigned char>& out,
546  const unsigned char* in, size_t& bp, size_t& pos, size_t inlength)
547  {
548  while((bp & 0x7) != 0) bp++; //go to first boundary of byte
549  size_t p = bp / 8;
550  if(p >= inlength - 4) { error = 52; return; } //error, bit pointer will jump past memory
551  unsigned long LEN = in[p] + 256u * in[p + 1], NLEN = in[p + 2] + 256u * in[p + 3]; p += 4;
552  if(LEN + NLEN != 65535) { error = 21; return; } //error: NLEN is not one's complement of LEN
553  if(p + LEN > inlength) { error = 23; return; } //error: reading outside of in buffer
554  for(unsigned long n = 0; n < LEN; n++)
555  {
556  out.push_back(in[p++]); //read LEN bytes of literal data
557  pos++;
558  }
559  bp = p * 8;
560  }

562  int decompress(std::vector<unsigned char>& out, const std::vector<unsigned char>& in) //returns error value
563  {
564  if(in.size() < 2) { return 53; } //error, size of zlib data too small
565  //error: 256 * in[0] + in[1] must be a multiple of 31, the FCHECK value is supposed to be made that way
566  if((in[0] * 256 + in[1]) % 31 != 0) { return 24; }
567  unsigned long CM = in[0] & 15, CINFO = (in[0] >> 4) & 15, FDICT = (in[1] >> 5) & 1;
568  //error: only compression method 8: inflate with sliding window of 32k is supported by the PNG spec
569  if(CM != 8 || CINFO > 7) { return 25; }
570  //error: the PNG spec says about the zlib stream: "The additional flags shall not specify a preset dictionary."
571  if(FDICT != 0) { return 26; }
572  inflate(out, in, 2);
573  return error; //note: adler32 checksum was skipped and ignored
574  }

577 struct ExtractPNG //PNG decoding and information extraction
578 {
579  std::vector<ZlibBlockInfo>* zlibinfo;
580  ExtractPNG(std::vector<ZlibBlockInfo>* info) : zlibinfo(info) {};
581  int error;
582  void decode(const unsigned char* in, size_t size)
583  {
584  error = 0;
585  if(size == 0 || in == 0) { error = 48; return; } //the given data is empty
586  readPngHeader(&in[0], size); if(error) return;
587  size_t pos = 33; //first byte of the first chunk after the header
588  std::vector<unsigned char> idat; //the data from idat chunks
589  bool IEND = false;
590  //loop through the chunks, ignoring unknown chunks and stopping at IEND chunk.
591  //IDAT data is put at the start of the in buffer
592  while(!IEND)
593  {
594  //error: size of the in buffer too small to contain next chunk
595  if(pos + 8 >= size) { error = 30; return; }
596  size_t chunkLength = read32bitInt(&in[pos]); pos += 4;
597  if(chunkLength > 2147483647) { error = 63; return; }
598  //error: size of the in buffer too small to contain next chunk
599  if(pos + chunkLength >= size) { error = 35; return; }
600  //IDAT chunk, containing compressed image data
601  if(in[pos + 0] == 'I' && in[pos + 1] == 'D' && in[pos + 2] == 'A' && in[pos + 3] == 'T')
602  {
603  idat.insert(idat.end(), &in[pos + 4], &in[pos + 4 + chunkLength]);
604  pos += (4 + chunkLength);
605  }
606  else if(in[pos + 0] == 'I' && in[pos + 1] == 'E' && in[pos + 2] == 'N' && in[pos + 3] == 'D')
607  {
608  pos += 4;
609  IEND = true;
610  }
611  else //it's not an implemented chunk type, so ignore it: skip over the data
612  {
613  pos += (chunkLength + 4); //skip 4 letters and uninterpreted data of unimplemented chunk
614  }
615  pos += 4; //step over CRC (which is ignored)
616  }
617  std::vector<unsigned char> out; //now the out buffer will be filled
618  ExtractZlib zlib(zlibinfo); //decompress with the Zlib decompressor
619  error = zlib.decompress(out, idat);
620  if(error) return; //stop if the zlib decompressor returned an error
621  }

623  //read the information from the header and store it in the Info
624  void readPngHeader(const unsigned char* in, size_t inlength)
625  {
626  if(inlength < 29) { error = 27; return; } //error: the data length is smaller than the length of the header
627  if(in[0] != 137 || in[1] != 80 || in[2] != 78 || in[3] != 71
628  || in[4] != 13 || in[5] != 10 || in[6] != 26 || in[7] != 10) { error = 28; return; } //no PNG signature
629  //error: it doesn't start with a IHDR chunk!
630  if(in[12] != 'I' || in[13] != 'H' || in[14] != 'D' || in[15] != 'R') { error = 29; return; }
631  }

633  unsigned long readBitFromReversedStream(size_t& bitp, const unsigned char* bits)
634  {
635  unsigned long result = (bits[bitp >> 3] >> (7 - (bitp & 0x7))) & 1;
636  bitp++;
637  return result;
638  }

641  {
642  unsigned long result = 0;
