789 bool cSimCharacter::BuildMultiBody(std::shared_ptr<cMultiBody>& out_body)
790 {
791  bool succ = true;
792

794  int num_joints = GetNumJoints();
795  bool fixed_base = FixedBase();
796  btVector3 base_intertia = btVector3(0, 0, 0);
797  btScalar base_mass = 0;
798  mMultBody = std::shared_ptr<cMultiBody>(new cMultiBody(num_joints, base_mass, base_intertia, fixed_base, false));
799 
800  btTransform base_trans;
801  base_trans.setIdentity();
802  mMultBody->setBaseWorldTransform(base_trans);
803 
804  for (int j = 0; j < num_joints; ++j)
805  {
806  cShape::eShape body_shape = cKinTree::GetBodyShape(mBodyDefs, j);
807  tVector body_size = cKinTree::GetBodySize(mBodyDefs, j);
808  double mass = cKinTree::GetBodyMass(mBodyDefs, j);
809  cKinTree::eJointType joint_type = cKinTree::GetJointType(mJointMat, j);
810  int parent_joint = cKinTree::GetParent(mJointMat, j);
811 
812  bool is_root = cKinTree::IsRoot(mJointMat, j);
813 
814  btCollisionShape* col_shape = BuildCollisionShape(body_shape, body_size);
815  btVector3 inertia = btVector3(0, 0, 0);
816  col_shape->calculateLocalInertia(static_cast<btScalar>(mass), inertia);
817 
818  // arg so many transforms...
819  tQuaternion this_to_parent = cMathUtil::EulerToQuaternion(cKinTree::GetAttachTheta(mJointMat, j));
820  tQuaternion body_to_this = cMathUtil::EulerToQuaternion(cKinTree::GetBodyAttachTheta(mBodyDefs, j));
821  tQuaternion this_to_body = body_to_this.inverse();
822 
823  tQuaternion parent_body_to_parent = tQuaternion::Identity();
824  if (parent_joint != gInvalidIdx)
825  {
826  parent_body_to_parent = cMathUtil::EulerToQuaternion(cKinTree::GetBodyAttachTheta(mBodyDefs, parent_joint));
827  }
828  tQuaternion parent_to_parent_body = parent_body_to_parent.inverse();
829 
830  tQuaternion body_to_parent_body = parent_to_parent_body * this_to_parent * body_to_this;
831  tQuaternion parent_body_to_body = body_to_parent_body.inverse();
832 
833  // parent body attachment point in body's coordinate frame
834  tVector parent_body_attach_pt = tVector::Zero();
835  if (parent_joint != gInvalidIdx)
836  {
837  parent_body_attach_pt = cKinTree::GetBodyAttachPt(mBodyDefs, parent_joint);
838  }
839  parent_body_attach_pt = cMathUtil::QuatRotVec(parent_to_parent_body, parent_body_attach_pt);
840 
841  tVector joint_attach_pt = cKinTree::GetAttachPt(mJointMat, j);
842  tVector body_attach_pt = cKinTree::GetBodyAttachPt(mBodyDefs, j);
843  joint_attach_pt = cMathUtil::QuatRotVec(parent_to_parent_body, joint_attach_pt);
844  joint_attach_pt -= parent_body_attach_pt;
845  body_attach_pt = cMathUtil::QuatRotVec(body_to_this.inverse(), body_attach_pt);
846 
847  btTransform parent_body_to_body_trans = btTransform::getIdentity();
848  parent_body_to_body_trans.setRotation(btQuaternion(parent_body_to_body.x(), parent_body_to_body.y(), parent_body_to_body.z(), parent_body_to_body.w()));
849 
850  bool disable_parent_collision = true;
851  if (is_root && !fixed_base)
852  {
853  mMultBody->setupFixed(j, static_cast<btScalar>(mass), inertia, parent_joint,
854  parent_body_to_body_trans.getRotation(),
855  world_scale * btVector3(joint_attach_pt[0], joint_attach_pt[1], joint_attach_pt[2]),
856  world_scale * btVector3(body_attach_pt[0], body_attach_pt[1], body_attach_pt[2]),
857  disable_parent_collision);
858  }
859  else
860  {
861  switch (joint_type)
862  {
863  case cKinTree::eJointTypeRevolute:
864  {
865  tVector axis = tVector(0, 0, 1, 0);
866  axis = cMathUtil::QuatRotVec(this_to_body, axis);
867 
868  mMultBody->setupRevolute(j, static_cast<btScalar>(mass), inertia, parent_joint,
869  parent_body_to_body_trans.getRotation(),
870  btVector3(axis[0], axis[1], axis[2]),
871  world_scale * btVector3(joint_attach_pt[0], joint_attach_pt[1], joint_attach_pt[2]),
872  world_scale * btVector3(body_attach_pt[0], body_attach_pt[1], body_attach_pt[2]),
873  disable_parent_collision);
874  }
875  break;
876  case cKinTree::eJointTypePlanar:
877  {
878  tVector offset = parent_body_attach_pt + joint_attach_pt
879  + cMathUtil::QuatRotVec(body_to_parent_body, body_attach_pt);
880 
881  tVector axis = tVector(0, 0, 1, 0);
882  axis = cMathUtil::QuatRotVec(this_to_body, axis);
883 
884  mMultBody->setupPlanar(j, static_cast<btScalar>(mass), inertia, parent_joint,
885  parent_body_to_body_trans.getRotation(),
886  btVector3(axis[0], axis[1], axis[2]),
887  world_scale * btVector3(offset[0], offset[1], offset[2]),
888  disable_parent_collision);
889  }
890  break;
891  case cKinTree::eJointTypePrismatic:
892  {
893  tVector axis = tVector(1, 0, 0, 0);
894  axis = cMathUtil::QuatRotVec(this_to_body, axis);
895 
896  mMultBody->setupPrismatic(j, static_cast<btScalar>(mass), inertia, parent_joint,
897  parent_body_to_body_trans.getRotation(),
898  btVector3(axis[0], axis[1], axis[2]),
899  world_scale * btVector3(joint_attach_pt[0], joint_attach_pt[1], joint_attach_pt[2]),
900  world_scale * btVector3(body_attach_pt[0], body_attach_pt[1], body_attach_pt[2]),
901  disable_parent_collision);
902  }
903  break;
904  case cKinTree::eJointTypeFixed:
905  {
906  mMultBody->setupFixed(j, static_cast<btScalar>(mass), inertia, parent_joint,
907  parent_body_to_body_trans.getRotation(),
908  world_scale * btVector3(joint_attach_pt[0], joint_attach_pt[1], joint_attach_pt[2]),
909  world_scale * btVector3(body_attach_pt[0], body_attach_pt[1], body_attach_pt[2]),
910  disable_parent_collision);
911  }
912  break;
913  case cKinTree::eJointTypeSpherical:
914  {
915  mMultBody->setupSpherical(j, static_cast<btScalar>(mass), inertia, parent_joint,
916  parent_body_to_body_trans.getRotation(),
917  world_scale * btVector3(joint_attach_pt[0], joint_attach_pt[1], joint_attach_pt[2]),
918  world_scale * btVector3(body_attach_pt[0], body_attach_pt[1], body_attach_pt[2]),
919  disable_parent_collision);
920  }
921  break;
922  default:
923  printf("Unsupported joint type");
924  assert(false);
925  break;
926  }
927  }
928 
929  btMultiBodyLinkCollider* col_obj = new btMultiBodyLinkCollider(mMultBody.get(), j);
930 
931  col_obj->setCollisionShape(col_shape);
932  btTransform col_obj_trans;
933  col_obj_trans.setIdentity();
934  col_obj->setWorldTransform(col_obj_trans);
935  col_obj->setFriction(mFriction);
936 
937  int collisionFilterGroup = GetPartColGroup(j);
938  int collisionFilterMask = GetPartColMask(j);
939  mWorld->AddCollisionObject(col_obj, collisionFilterGroup, collisionFilterMask);
940  mMultBody->getLink(j).m_collider = col_obj;
941  }
942 
943  mMultBody->finalizeMultiDof();
944 
945  return succ;
946 }
