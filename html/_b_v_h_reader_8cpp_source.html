480 tMatrix cBVHReader::getTranslationForFrame(size_t channel_start, const std::vector<Channels> & channels_order, const Eigen::MatrixXd & frame_data,
481  size_t frame) const
482 {
483  tMatrix mat = tMatrix::Identity();
484 
485  for (size_t i = 0; i < channels_order.size(); i++)
486  {
487  Channels curr_channel = channels_order[i];
488  double val = frame_data(frame, channel_start + i);
489  if (curr_channel == Xposition)
490  {
491  mat(0, 3) = val;
492  }
493  else if (curr_channel == Yposition)
494  {
495  mat(1, 3) = val;
496  }
497  else if (curr_channel == Zposition)
498  {
499  mat(2, 3) = val;
500  }
501  }
502 
503  mat(3, 3) = 0; // ignore translation for local transforms
504  mat = mModelTrans * mat;
505  mat(3, 3) = 1;
506  return mat;
507 }

598 cKinTree::eJointType cBVHReader::CalcJointType(const tJointData& joint_data) const
599 {
600  cKinTree::eJointType joint_type = cKinTree::eJointTypeNone;
601  if (joint_data.parent == gInvalidIdx)
602  {
603  joint_type = cKinTree::eJointTypeNone;
604  }
605  else
606  {
607  unsigned int channel_mask = 0;
608  for (size_t i = 0; i < joint_data.channels_order.size(); ++i)
609  {
610  Channels curr_channel = joint_data.channels_order[i];
611  channel_mask |= 1 << curr_channel;
612  }
613 
614  switch (channel_mask)
615  {
616  case 0:
617  joint_type = cKinTree::eJointTypeFixed;
618  break;
619  case (1 << Xrotation) :
620  case (1 << Yrotation) :
621  case (1 << Zrotation) :
622  joint_type = cKinTree::eJointTypeRevolute;
623  break;
624  case (1 << Xposition) :
625  case (1 << Yposition) :
626  case (1 << Zposition) :
627  joint_type = cKinTree::eJointTypePrismatic;
628  break;
629  case (1 << Xrotation) | (1 << Yrotation) | (1 << Zrotation) :
630  joint_type = cKinTree::eJointTypeSpherical;
631  break;
632  case (1 << Xposition) | (1 << Yposition) :
633  case (1 << Xposition) | (1 << Zposition) :
634  case (1 << Yposition) | (1 << Zposition) :
635  joint_type = cKinTree::eJointTypePlanar;
636  break;
637  case (1 << Xrotation) | (1 << Yrotation) | (1 << Zrotation)
638  | (1 << Xposition) | (1 << Yposition) | (1 << Zposition) :
639  joint_type = cKinTree::eJointTypeSpherical; // hack
640  break;
641  default:
642  assert(false); // unsupported joint type
643  break;
644  }
645  }
646 
647  return joint_type;
648 }
