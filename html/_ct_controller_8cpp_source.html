368 void cCtController::BuildStatePose(Eigen::VectorXd& out_pose) const
369 {
370  tMatrix origin_trans = mChar->BuildOriginTrans();
371  tQuaternion origin_quat = cMathUtil::RotMatToQuaternion(origin_trans);
372 
373  bool flip_stance = FlipStance();
374  if (flip_stance)
375  {
376  origin_trans.row(2) *= -1; // reflect z
377  }
378 
379  tVector root_pos = mChar->GetRootPos();
380  tVector root_pos_rel = root_pos;
381 
382  root_pos_rel[3] = 1;
383  root_pos_rel = origin_trans * root_pos_rel;
384  root_pos_rel[3] = 0;
385 
386  out_pose = Eigen::VectorXd::Zero(GetStatePoseSize());
387  out_pose[0] = root_pos_rel[1];
388  int num_parts = mChar->GetNumBodyParts();
389  int root_id = mChar->GetRootID();
390 
391  int pos_dim = GetPosFeatureDim();
392  int rot_dim = GetRotFeatureDim();
393 
394  tQuaternion mirror_inv_origin_quat = origin_quat.conjugate();
395  mirror_inv_origin_quat = cMathUtil::MirrorQuaternion(mirror_inv_origin_quat, cMathUtil::eAxisZ);
396 
397  int idx = 1;
398  for (int i = 0; i < num_parts; ++i)
399  {
400  int part_id = RetargetJointID(i);
401  if (mChar->IsValidBodyPart(part_id))
402  {
403  const auto& curr_part = mChar->GetBodyPart(part_id);
404  tVector curr_pos = curr_part->GetPos();
405 
406  if (mRecordWorldRootPos && i == root_id)
407  {
408  if (flip_stance)
409  {
410  curr_pos = cMathUtil::QuatRotVec(origin_quat, curr_pos);
411  curr_pos[2] = -curr_pos[2];
412  curr_pos = cMathUtil::QuatRotVec(mirror_inv_origin_quat, curr_pos);
413  }
414  }
415  else
416  {
417  curr_pos[3] = 1;
418  curr_pos = origin_trans * curr_pos;
419  curr_pos -= root_pos_rel;
420  curr_pos[3] = 0;
421  }
422 
423  out_pose.segment(idx, pos_dim) = curr_pos.segment(0, pos_dim);
424  idx += pos_dim;
425 
426  tQuaternion curr_quat = curr_part->GetRotation();
427  if (mRecordWorldRootRot && i == root_id)
428  {
429  if (flip_stance)
430  {
431  curr_quat = origin_quat * curr_quat;
432  curr_quat = cMathUtil::MirrorQuaternion(curr_quat, cMathUtil::eAxisZ);
433  curr_quat = mirror_inv_origin_quat * curr_quat;
434  }
435  }
436  else
437  {
438  curr_quat = origin_quat * curr_quat;
439  if (flip_stance)

442  }
443  }
444
