225 void cPDController::GetTargetTheta(Eigen::VectorXd& out_theta) const
226 {
227  const cSimBodyJoint& joint = GetJoint();
228  cKinTree::eJointType joint_type = joint.GetType();
229  out_theta = mParams.segment(eParamTargetTheta0, GetJointDim());
230 
231  if (UseWorldCoord() && joint_type == cKinTree::eJointTypeRevolute)
232  {
233  tVector axis_world;
234  double theta_world;
235  joint.CalcWorldRotation(axis_world, theta_world);
236 
237  Eigen::VectorXd pose;
238  joint.BuildPose(pose);
239  double theta_rel = pose[0];
240  tVector axis_rel = joint.GetAxisRel();
241 
242  tVector axis_ref = joint.CalcAxisWorld();
243  double theta_offset = axis_world.dot(axis_ref) * theta_world
244  - axis_rel.dot(axis_ref) * theta_rel;
245  out_theta(0) -= theta_offset;
246  }
247 
248  if (UseWorldCoord() && joint_type == cKinTree::eJointTypeSpherical)
249  {
250  Eigen::VectorXd pose;
251  joint.BuildPose(pose);
252 
253  tQuaternion world_rot = joint.CalcWorldRotation();
254  tQuaternion rel_rot = cMathUtil::VecToQuat(pose);
255 
256  tQuaternion tar_q = cMathUtil::VecToQuat(out_theta);
257  tQuaternion diff = world_rot.conjugate() * tar_q; // diference between target world and world
258  tar_q = rel_rot * diff; // Add this difference to current relative rotation
259  out_theta = cMathUtil::QuatToVec(tar_q);
260  }
261
