542 void cSceneSimChar::ResolveCharGroundIntersect(const std::shared_ptr<cSimCharacter>& out_char) const
543 {
544  const double pad = 0.001;
545 
546  int num_parts = out_char->GetNumBodyParts();
547  double min_violation = 0;
548  for (int b = 0; b < num_parts; ++b)
549  {
550  if (out_char->IsValidBodyPart(b))
551  {
552  tVector aabb_min;
553  tVector aabb_max;
554  const auto& part = out_char->GetBodyPart(b);
555  part->CalcAABB(aabb_min, aabb_max);
556 
557  tVector mid = 0.5 * (aabb_min + aabb_max);
558  tVector sw = tVector(aabb_min[0], 0, aabb_min[2], 0);
559  tVector nw = tVector(aabb_min[0], 0, aabb_max[2], 0);
560  tVector ne = tVector(aabb_max[0], 0, aabb_max[2], 0);
561  tVector se = tVector(aabb_max[0], 0, aabb_min[2], 0);
562 
563  double max_ground_height = 0;
564  max_ground_height = mGround->SampleHeight(aabb_min);
565  max_ground_height = std::max(max_ground_height, mGround->SampleHeight(mid));
566  max_ground_height = std::max(max_ground_height, mGround->SampleHeight(sw));
567  max_ground_height = std::max(max_ground_height, mGround->SampleHeight(nw));
568  max_ground_height = std::max(max_ground_height, mGround->SampleHeight(ne));
569  max_ground_height = std::max(max_ground_height, mGround->SampleHeight(se));
570  max_ground_height += pad;
571 
572  double min_height = aabb_min[1];
573  min_violation = std::min(min_violation, min_height - max_ground_height);
574  }
575  }
576 
577  if (min_violation < 0)
578  {
579  tVector root_pos = out_char->GetRootPos();
580  root_pos[1] += -min_violation;
581  out_char->SetRootPos(root_pos);

886 void cSceneSimChar::SpawnProjectile(double density, double min_size, double max_size,
887  double min_speed, double max_speed, double y_offset,
888  double life_time)
889 {
890  double min_dist = 1;
891  double max_dist = 2;
892  tVector aabb_min;
893  tVector aabb_max;
894 
895  int char_id = mRand.RandInt(0, GetNumChars());
896  const auto& curr_char = GetCharacter(char_id);
897  curr_char->CalcAABB(aabb_min, aabb_max);
898 
899  tVector aabb_center = (aabb_min + aabb_max) * 0.5;
900  tVector obj_size = tVector(1, 1, 1, 0) * mRand.RandDouble(min_size, max_size);
901 
902  double rand_theta = mRand.RandDouble(0, M_PI);
903  double rand_dist = mRand.RandDouble(min_dist, max_dist);
904 
905  double aabb_size_x = (aabb_max[0] - aabb_min[0]);
906  double aabb_size_z = (aabb_max[2] - aabb_min[2]);
907  double buffer_dist = std::sqrt(aabb_size_x * aabb_size_x + aabb_size_z * aabb_size_z);
908 
909  double rand_x = 0.5 * buffer_dist + rand_dist * std::cos(rand_theta);
910  rand_x *= mRand.RandSign();
911  rand_x += aabb_center[0];
912  double rand_y = mRand.RandDouble(aabb_min[1], aabb_max[1]) + obj_size[1] * 0.5;
913  rand_y += y_offset;
914 
915  double rand_z = aabb_center[2];
916  rand_z = 0.5 * buffer_dist + rand_dist * std::sin(rand_theta);
917  rand_z *= mRand.RandSign();
918  rand_z += aabb_center[2];
919 
920  tVector pos = tVector(rand_x, rand_y, rand_z, 0);
921  tVector target = tVector(mRand.RandDouble(aabb_min[0], aabb_max[0]),
922  mRand.RandDouble(aabb_min[1], aabb_max[1]), aabb_center[2], 0);
923 
924  tVector com_vel = curr_char->CalcCOMVel();
925  tVector vel = (target - pos).normalized();
926  vel *= mRand.RandDouble(min_speed, max_speed);
927  vel[0] += com_vel[0];
928  vel[2] += com_vel[2];
929 
930  cSimBox::tParams params;
931  params.mSize = obj_size;
932  params.mPos = pos;
933  params.mVel = vel;
934  params.mFriction = 0.7;
935  params.mMass = density * params.mSize[0] * params.mSize[1] * params.mSize[2];
936  std::shared_ptr<cSimBox> box = std::shared_ptr<cSimBox>(new cSimBox());
937  box->Init(mWorld, params);
938  box->UpdateContact(cWorld::eContactFlagObject, cContactManager::gFlagNone);
939 
940  tObjEntry obj_entry;
941  obj_entry.mObj = box;
942  obj_entry.mEndTime = GetTime() + life_time;
943 
944  AddObj(obj_entry);
945 }
